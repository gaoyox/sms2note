3）写入小米笔记（核心，就几行）
java
运行
void insertToMiuiNotes(Context context, String title, String content) {
    ContentValues cv = new ContentValues();
    cv.put("title", title);
    cv.put("content", content);
    cv.put("created_time", System.currentTimeMillis());

    Uri uri = Uri.parse("content://com.miui.notes/note");
    context.getContentResolver().insert(uri, cv);
}
